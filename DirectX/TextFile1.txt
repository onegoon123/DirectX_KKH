	static bool Fast = false;
	Fast = GameEngineInput::IsPress("ButtonB");

	if (GameEngineInput::IsDown("Down") || (GameEngineInput::IsPress("Down") && Fast))
	{
		PortraitRender->GetTransform()->AddLocalPosition(float4::Down);
	}
	if (GameEngineInput::IsDown("Right") || (GameEngineInput::IsPress("Right") && Fast))
	{
		PortraitRender->GetTransform()->AddLocalPosition(float4::Right);
	}
	if (GameEngineInput::IsDown("Left") || (GameEngineInput::IsPress("Left") && Fast))
	{
		PortraitRender->GetTransform()->AddLocalPosition(float4::Left);
	}
	if (GameEngineInput::IsDown("Up") || (GameEngineInput::IsPress("Up") && Fast))
	{
		PortraitRender->GetTransform()->AddLocalPosition(float4::Up);
	}
	if (GameEngineInput::IsDown("ButtonR"))
	{
		MsgTextBox(PortraitRender->GetTransform()->GetLocalPosition().ToString());
	}
